# Check Empty Object

Various methods to check for empty Object.

# JS

# ES8 (2017)

const data = {};
Object.entries(data).length === 0 && data.constructor === Object; // true

Note: Object.entries not supported in IE browsers.

# ES6 (2015)

const data = {};
Object.keys(data).length === 0 && data.constructor === Object; // true

# ES5 (Standard)

var data = {};
function isObjectEmpty(data) {
  for (var prop in data) {
    if(data.hasOwnProperty(prop)) {
      return false;
    }
  }
  return true;
}
isObjectEmpty(); // true

# jQuery

const data = {};
jQuery.isEmptyObject(data); // true

# Lodash

const data = {};
_.isEmpty(data); // true
